inventory_analysis.sql
SELECT vendor_id, product_sku,
       lead_time_days, reorder_point,
       SUM(qty_on_hand) AS total_stock,
       AVG(oee_score) AS avg_oee
FROM supply_chain_ops sc
JOIN vendor_master vm ON sc.vendor_id = vm.id
WHERE lead_time_days < 20
  AND oee_score > 0.80
GROUP BY vendor_id, product_sku
ORDER BY avg_oee DESC;
